Работа с базой данных
Задание 1

SELECT login,
       COUNT(*) AS quantity
FROM "Orders"
INNER JOIN "Couriers" AS c ON "courierId" = c.id
WHERE "inDelivery" = true
GROUP BY login;

Задание 2

SELECT track,
       CASE
           WHEN finished = true THEN 2
           WHEN cancelled = true THEN -1
           WHEN "inDelivery" = true THEN 1
           ELSE 0
       END AS status
FROM "Orders";
